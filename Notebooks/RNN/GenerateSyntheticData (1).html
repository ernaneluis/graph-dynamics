def plot_clustering(data, clustering_model, pieces, is_normalize=False, min_x=1, max_x=365., save_to_file=None):
    total = len(data)
    xcoords = np.arange(min_x, max_x, pieces)
    for j, centroid in enumerate(clustering_model['centroids']):
        cluster_size = clustering_model['cluster_size'][j]
#         plt.title('{0:.2%}'.format(cluster_size/float(total)), fontsize=26)
        for i in range(cluster_size):
            plt.plot(np.arange(min_x, max_x), data[clustering_model['labels'] == j][i], alpha=.6, color='#168EF7')
        if is_normalize:
            centroid = normalize(0., 1., centroid)
        #plt.tight_layout()
        plt.gcf().subplots_adjust(bottom=0.15)
        plt.xlim(min_x, max_x)
        plt.xticks([], fontsize=14)
        plt.yticks([], fontsize=14)
#         plt.gca().spines['top'].set_visible(False)
#         plt.gca().spines['right'].set_visible(False)
        plt.plot(np.arange(0, len(centroid)), centroid, linewidth=3.0, color='#F88017')
        for xc in xcoords:
            plt.axvline(x=xc, linestyle='--', linewidth=.5)

        if not save_to_file is None:
            plt.savefig( "{1}_cluster_{0}.pdf".format(j, save_to_file))
            plt.close()
        else:
            plt.show()

def normalize(low=0, high=1, A=None):
    mins = np.min(A, axis=0)
    maxs = np.max(A, axis=0)
    rng = maxs - mins
    return high - (((high - low) * (maxs - A)) / rng)
